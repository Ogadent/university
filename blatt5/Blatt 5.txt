	int aufsteigendeSotierung(int a[], int s){
	int i,j,k; =0
	while(i!>j && k<s){
	i = a[k];
	j = a[k+1];
	k++;
	}
	if(k>=s){
	i = a[k-1];
	j = a[k];
	if(i!>j){
	return 1;
	}
	}
	return 0;
	}

	int kleinsterWert(int a[], int s){
	int i,j,k; = 0
	i= a[k];
	j= a[k+1];
	while(k<s){
	if(i <= j){
	k++;
	j= a[k+1];
	}
	else{
	i = j;
	k++;
	j= a[k+1];
	}
	}
	return i;
	}
